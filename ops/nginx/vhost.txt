# $GA_APP_URL (docker)
server {
  server_name $GA_APP_URL;

  location / {
    proxy_pass $GA_NGX_PROXY;
    include /etc/nginx/proxy_params;
  }

  listen [::]:443 ssl http2;
  listen 443 ssl http2;
  include /etc/nginx/global/security.conf;
  include /etc/nginx/global/ssl.conf; 
  ssl_certificate $NGX_CERT;
  ssl_certificate_key $NGX_CERT_KEY;
}
