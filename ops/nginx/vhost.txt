# $APP_URL (docker)
server {
  server_name $APP_URL;

  location / {
    proxy_pass $APP_PROXY_URL;
    include /etc/nginx/proxy_params;
  }

  listen [::]:443 ssl http2;
  listen 443 ssl http2;
  include /etc/nginx/global/security.conf;
  include /etc/nginx/global/ssl.conf; 
  ssl_certificate $CERT;
  ssl_certificate_key $CERT_KEY;
}
